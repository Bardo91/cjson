175  bool Json::operator==(const Json& _x) const {
176  if(mType != _x.mType)
177  return false;
178  switch (mType)
179  {
180  case DataType::null:
181  return true;
182  case DataType::text:
183  return mText == _x.mText;
184  case DataType::array:
185  if(size() != _x.size())
186  return false;
187  for(size_t i = 0; i < size(); ++i) {
188  if(!(*mArray[i] == *_x.mArray[i]))
189  return false;
190  }
191  return true;
192  case DataType::object:
193  if(size() != _x.size())
194  return false;
195  for(const auto& myElement : mObject) {
196  const std::string& key = myElement.first;
197  if(!_x.contains(key))
198  return false;
199  if(!(_x[key] == *myElement.second))
200  return false;
201  }
202  return true;
203  default:
204  return mNumber.i == _x.mNumber.i;
205  }
206  }
